Ejemplo 1: array nulo

{{valoresNulo | join}}

Ejemplo 2: Valores separados con , (no indicando el parametro)

{{valores | join}}

Ejemplo 3: Valores separados con |

{{valores | join: '|'}}

Como instalar

      
npm i ddr-join-pipe

Como importar

      
// Angular
import { BrowserModule } from '@angular/platform-browser';
import { NgModule } from '@angular/core';

// Modules
import { AppRoutingModule } from './app-routing.module';
import { DdrJoinPipeModule } from 'ddr-join-pipe';

@NgModule({
  declarations: [
    AppComponent,
    SampleDdrJoinPipeComponent
  ],
  imports: [
    BrowserModule,
    AppRoutingModule,
    DdrJoinPipeModule
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule { }
