TS

      
export class DdrSpinnerDemo {

  public embedded: boolean;

  constructor(private ddrSpinner: DdrSpinnerService) { 
    this.embedded = false;
  }

  activarSpinner(){
    this.ddrSpinner.showSpinner();

    setTimeout(() => {
      this.ddrSpinner.hideSpinner();
    }, 50000);
  }

  cambiarModo(){
    this.embedded = !this.embedded;
  }

}
